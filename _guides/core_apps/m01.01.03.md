## Introduction

With the book shape and the `Map`-based storage, the four CRUD operations are thin, focused functions in `inventory.js`. None of them touch the filesystem directly as they take an `inventory` (a `Map`) and return a result.

## The Four Operations

```js
function createBook(inventory, book) { /* ... */ }
function getBook(inventory, isbn) { /* ... */ }
function listBooks(inventory) { /* ... */ }
function updateBook(inventory, isbn, changes) { /* ... */ }
function deleteBook(inventory, isbn) { /* ... */ }
```

## Create

```js
function createBook(inventory, book) {
  if (!isValidBook(book)) return { ok: false, error: "invalid book" };
  if (inventory.has(book.isbn)) return { ok: false, error: "isbn already exists" };
  inventory.set(book.isbn, { ...book, addedAt: new Date().toISOString() });
  return { ok: true, book: inventory.get(book.isbn) };
}
```

## Read

```js
function getBook(inventory, isbn) {
  const book = inventory.get(isbn);
  return book ? { ...book } : null;
}

function listBooks(inventory) {
  return Array.from(inventory.values())
    .map(book => ({ ...book }))
    .sort((a, b) => a.title.localeCompare(b.title));
}
```

## Update

```js
function updateBook(inventory, isbn, changes) {
  if (!inventory.has(isbn)) return { ok: false, error: "not found" };
  const { isbn: _ignored, ...safeChanges } = changes;
  const updated = { ...inventory.get(isbn), ...safeChanges };
  inventory.set(isbn, updated);
  return { ok: true, book: { ...updated } };
}
```

## Delete

```js
function deleteBook(inventory, isbn) {
  if (!inventory.has(isbn)) return { ok: false, error: "not found" };
  const removed = inventory.get(isbn);
  inventory.delete(isbn);
  return { ok: true, book: removed };
}
```

## Wiring It to Persistence

`inventory.js` never calls `saveRaw()` itself as that decision belongs to whatever calls these functions:

```js
// index.js
const { loadRaw, saveRaw } = require("./src/storage");
const { loadInventory, serializeInventory } = require("./src/inventory");
const { createBook } = require("./src/inventory");

const inventory = loadInventory(loadRaw());
const result = createBook(inventory, {
  isbn: "978-0134685991",
  title: "Effective Java",
  author: "Joshua Bloch"
});
if (result.ok) saveRaw(serializeInventory(inventory));
```

## Programming Tips

- Return plain result objects (`{ ok, error }` or `{ ok, book }`) instead of throwing for expected outcomes like "not found" or "duplicate ISBN." Save exceptions for actual I/O failures; the bonus front-end module needs to *display* "not found," not catch an exception for it.
- Before building any UI, write a small script that creates, reads, updates, and deletes one book end to end and logs each result. If that script doesn't work cleanly, no front-end will fix it.
