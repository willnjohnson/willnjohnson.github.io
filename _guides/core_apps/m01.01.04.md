## Introduction

Everything so far runs in Node. This bonus module keeps the exact same `createBook` / `getBook` / `listBooks` / `updateBook` / `deleteBook` functions from the previous sections and drives them from a browser page instead of a script. The only piece that changes is persistence, since a browser can't touch `data/books.json` directly.

## Bridging JS Logic to a Browser

If `inventory.js` has no Node-specific imports (i.e. no `fs`, no `path`, or no `require` for anything but plain JS) you can load the exact same file in a `<script>` tag and the CRUD functions work unchanged in the browser.

## Browser Storage Layer

```js
// src/storage.browser.js
const STORAGE_KEY = "book-inventory";

function loadRaw() {
  const raw = localStorage.getItem(STORAGE_KEY);
  return raw ? JSON.parse(raw) : [];
}

function saveRaw(rawArray) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(rawArray));
}
```

WE use the same two function names and same signatures as `storage.js`. For `inventory.js` and everything calling it, we don't know or care that the backing store changed from a file to `localStorage`.

## Rendering the List

```js
// src/app.js
function renderBooks(inventory) {
  const container = document.getElementById("book-list");
  container.innerHTML = "";
  for (const book of listBooks(inventory)) {
    const row = document.createElement("div");
    row.className = "book-row";
    row.innerHTML = `
      <img src="${book.image || "placeholder.png"}" onerror="this.src='placeholder.png'">
      <span>${book.title} — ${book.author} (${book.isbn})</span>
      <button data-isbn="${book.isbn}" class="delete-btn">Delete</button>
    `;
    container.appendChild(row);
  }
}
```

## Wiring Forms to CRUD

```js
let inventory = loadInventory(loadRaw());
renderBooks(inventory);

document.getElementById("book-form").addEventListener("submit", (e) => {
  e.preventDefault();
  const data = Object.fromEntries(new FormData(e.target));
  const result = createBook(inventory, data);
  if (result.ok) {
    saveRaw(serializeInventory(inventory));
    renderBooks(inventory);
    e.target.reset();
  } else {
    alert(result.error);
  }
});

document.getElementById("book-list").addEventListener("click", (e) => {
  if (!e.target.classList.contains("delete-btn")) return;
  deleteBook(inventory, e.target.dataset.isbn);
  saveRaw(serializeInventory(inventory));
  renderBooks(inventory);
});
```
